# leetcode 133 图复制

简单图： 没有平行边， 也没有自环。

要求从给出某个节点开始， 提供一个图的复制。 

本质是一个图的遍历过程。 



遍历一遍图， 每到一个点， 

1. 判断这个点是否已经复制过。（借由一个<node* old,  node* copy> map）

2. 复制这个点的连通关系

   创建一个新vec， 把old.neighbers 复制进去。





基于递归的深度搜索：

~~~C++
/*
// Definition for a Node.
class Node {
public:
    int val;
    vector<Node*> neighbors;

    Node() {}

    Node(int _val, vector<Node*> _neighbors) {
        val = _val;
        neighbors = _neighbors;
    }
};
*/
class Solution {
public:
    unordered_map<Node*, Node*> mapping;
    Node* cloneGraph(Node* node) {
        //空
        if(node == NULL) {return NULL;}
        //已经复制过， 返回复制好的节点
        if(mapping.find(node) != mapping.end()){
            return mapping[node];
        }
        // 复制连通关系
        vector<Node*> new_neigh;
        Node* root = new Node(node->val, new_neigh);
        mapping[node] = root;
        // dfs
        for(auto n : node->neighbors) {
           root->neighbors.push_back(cloneGraph(n));
        }
        
        return root;
    }
};
~~~

~~~c++
class Solution {
public:
    unordered_map<Node*, Node*> mapping;
    Node* cloneGraph(Node* node) {
        //空
        if(node == NULL) {return NULL;}
        //bfs 找到所有节点
        set<Node*> gra;
        queue<Node*> bfs_a;
        bfs_a.push(node);
        gra.insert(node);

        while (!bfs_a.empty()) {
            Node* r = bfs_a.front();
            bfs_a.pop();
            //bfs
            for(auto i : r->neighbors) {
                if (gra.find(i) == gra.end()) {
                    gra.insert(i);
                    bfs_a.push(i);
                }
            }
        }

        // 复制所有节点
        unordered_map<Node*, Node*> old_copy;
        for(auto no : gra) {
            vector<Node*> nei;
            Node* n_node = new Node(no->val, nei);
            old_copy[no] = n_node;
        }


        // 复制连通关系
        for(auto n : gra) {
            for(auto neigh : n->neighbors) {
                old_copy[n]->neighbors.push_back(old_copy[neigh]);
            }
        }
        
        return old_copy[node];
    }
};
~~~
